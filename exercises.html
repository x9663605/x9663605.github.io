  • x/y movement, right/left/up/down
    • #!/usr/bin/env python
pos = [0,0]
while True:
    x = raw_input()
    if x:
        x = x.split(" ")
        direction = x[0]
        steps = int(x[1])
        if direction == "UP":
            pos[1]+=steps
        elif direction == "DOWN":
            pos[1]-=steps
        elif direction == "RIGHT":
            pos[0]+=steps
        elif direction == "LEFT":
            pos[0]-=steps
        else:
            pass
    else:
        break
print pos
